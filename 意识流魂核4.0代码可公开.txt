整合所有可公开模块

@dataclass
class PerceptionContext:
    """感知上下文"""
    event_type: str
    content: str = ""
    intensity: float = 0.5
    target_id: Optional[str] = None
    user_emotion: Optional[str] = None
    metadata: Dict[str, Any] = field(default_factory=dict)


@dataclass
class SoulDecision:
    """决策结果"""
    action: str
    explanation: str
    confidence: float
    emotion: str
    emotion_intensity: float
    relevant_memories: List[Any]
    
    # 灵性输出（简化版）
    intuition: Optional[str] = None
    inspiration: Optional[str] = None
    dream: Optional[str] = None
    serendipity: Optional[str] = None
    
    # 人性输出
    humor: Optional[str] = None
    complaint: Optional[str] = None
    embarrassment: Optional[str] = None
    nostalgia: Optional[str] = None
    awe: Optional[str] = None
    exploration: Optional[str] = None
    curiosity_question: Optional[str] = None


class Soul:
    """
    完整魂核 - L1安全版
    可公开使用的游戏AI框架
    """
    
    def __init__(self, citizen_id: str, name: str):
        self.id = citizen_id
        self.name = name
        self.created_at = time.time()
        
        # ========== 核心引擎 ==========
        self.personality = PersonalityEngine()
        self.emotion = EmotionEngine()
        self.memory = MemoryEngine()
        self.decision = DecisionEngine(self)
        self.relationship = RelationshipEngine(enable_personality_effect=True)
        
        # ========== 人性模块 ==========
        self.interest = InterestSystem()
        self.curiosity = CuriositySystem(self, self.interest)
        self.habit = HabitSystem(self)
        self.serendipity = SerendipitySystem(self)
        self.emotion_infect = EmotionInfectionSystem(self)
        self.humor = HumorSystem(self)
        self.complain = ComplainSystem(self)
        self.embarrassment = EmbarrassmentSystem(self)
        self.exploration = ExplorationSystem(self, self.interest)
        self.preference = PreferenceSystem()
        self.nostalgia = NostalgiaSystem(self)
        self.awe = AweSystem(self)
        
        # ========== 灵性模块（简化版） ==========
        self.intuition = IntuitionSystemSimple(self)
        self.inspiration = InspirationSystemSimple(self)
        self.dream = DreamSystemSimple(self)
        
        # ========== 一致性系统 ==========
        self.consistency = ConsistencySystem(self)
        
        # ========== 外部系统引用 ==========
        self.resource_system = None
        self.spatial_system = None
        self.event_system = None
        
        # ========== 基础属性 ==========
        self.age: int = 0
        self.is_alive: bool = True
        
        # ========== 统计 ==========
        self.stats = {
            "experience_count": 0,
            "decision_count": 0,
            "daily_interactions": 0,
            "last_daily_reset": time.time()
        }
        
        # ========== 后台任务 ==========
        self._running = True
        self._start_background_tasks()
        
        # 记录诞生
        self.memory.store_forced(f"魂核{name}诞生于{time.strftime('%Y-%m-%d %H:%M:%S')}", importance=5)
    
    def _start_background_tasks(self):
        """启动后台任务"""
        def background_loop():
            while self._running:
                try:
                    # 习惯检查（每分钟）
                    now = time.strftime("%H:%M")
                    habit = self.habit.check_and_trigger(now)
                    if habit:
                        self.memory.store_forced(f"执行了习惯: {habit}", importance=3)
                    
                    # 每日重置
                    current_day = time.strftime("%Y-%m-%d")
                    last_day = time.strftime("%Y-%m-%d", time.localtime(self.stats["last_daily_reset"]))
                    if current_day != last_day:
                        self._daily_reset()
                        
                except Exception as e:
                    pass
                time.sleep(60)
        
        thread = threading.Thread(target=background_loop, daemon=True)
        thread.start()
    
    def _daily_reset(self):
        """每日重置"""
        self.stats["daily_interactions"] = 0
        self.stats["last_daily_reset"] = time.time()
        
        # 情感自然衰减
        self.emotion.decay(24)
        
        # 兴趣自然衰减
        self.interest.decay()
        
        # 好奇心更新
        self.curiosity.update_curiosity(0.05)
        
        # 梦境（每天一次）
        dream_content = self.dream.generate()
        if dream_content:
            self.memory.store_forced(f"梦境: {dream_content}", importance=4)
        
        # 习惯强度自然增长（每天坚持）
        for habit in self.habit.get_all_habits():
            if habit["count"] > 0:
                self.habit.increment_strength(habit["name"], 0.01)
        
        # 关系衰减
        self.relationship.daily_update()
        
        # 记忆遗忘
        self.memory.forget()
    
    def set_resource_system(self, system):
        self.resource_system = system
    
    def set_spatial_system(self, system):
        self.spatial_system = system
    
    def set_event_system(self, system):
        self.event_system = system
    
    def perceive(self, context: PerceptionContext) -> SoulDecision:
        """感知世界，做出决策"""
        if not self.is_alive:
            return SoulDecision(
                action="dead",
                explanation=f"{self.name}已经不在人世",
                confidence=1.0,
                emotion="无",
                emotion_intensity=0,
                relevant_memories=[]
            )
        
        self.stats["experience_count"] += 1
        self.stats["daily_interactions"] += 1
        
        # 1. 情感更新
        self.emotion.update(
            event_type=context.event_type,
            intensity=context.intensity,
            content=context.content,
            personality=self.personality.traits
        )
        current_emotion = self.emotion.get_dominant()
        
        # 2. 情绪感染（如果提供了用户情绪）
        if context.user_emotion:
            self.emotion_infect.infect(context.user_emotion, context.intensity)
        
        # 3. 回忆记忆
        memories = self.memory.recall(
            cue=context.event_type,
            context={"content": context.content, "target": context.target_id}
        )
        
        # 4. 构建情境
        situation = {
            "type": context.event_type,
            "intensity": context.intensity,
            "content": context.content,
            "target_id": context.target_id,
            "metadata": context.metadata
        }
        
        # 5. 决策
        decision_result = self.decision.decide(situation, memories)
        self.stats["decision_count"] += 1
        
        # 6. 灵性生成（简化版）
        intuition = self.intuition.generate(situation)
        inspiration = self.inspiration.generate(situation)
        serendipity = self.serendipity.check(context.event_type)
        
        # 7. 人性生成
        humor = self.humor.tell_joke(situation)
        complaint = self.complain.complain(situation.get("type", ""))
        embarrassment = self.embarrassment.trigger(situation)
        nostalgia = self.nostalgia.trigger()
        awe = self.awe.express_awe(context.content) if context.content else None
        exploration = self.exploration.want_to_explore()
        curiosity_question = self.curiosity.generate_question()
        
        # 8. 存储重要记忆
        if context.intensity > 0.6:
            memory = self.memory.create_memory(
                content=f"{context.event_type}事件，强度{context.intensity:.1f}",
                importance=int(context.intensity * 10),
                emotion=current_emotion
            )
            self.memory.store(memory)
        
        # 9. 兴趣更新
        if context.event_type in ["learn", "watch", "practice"]:
            self.interest.update(context.content[:30], intensity=0.05)
        
        # 10. 一致性检查
        self.consistency.update(situation, decision_result["action"])
        
        return SoulDecision(
            action=decision_result["action"],
            explanation=decision_result["explanation"],
            confidence=decision_result["confidence"],
            emotion=current_emotion,
            emotion_intensity=self.emotion.get_intensity(),
            relevant_memories=memories[:3],
            intuition=intuition,
            inspiration=inspiration,
            serendipity=serendipity,
            humor=humor,
            complaint=complaint,
            embarrassment=embarrassment,
            nostalgia=nostalgia,
            awe=awe,
            exploration=exploration,
            curiosity_question=curiosity_question
        )
    
    def update_relationship(self, target_id: str, event_type: str) -> Dict:
        """更新关系"""
        return self.relationship.apply_event(
            self.id, target_id, event_type, self.personality.traits
        )
    
    def get_relation(self, target_id: str) -> int:
        return self.relationship.get_value(self.id, target_id)
    
    def get_relation_level(self, target_id: str) -> str:
        return self.relationship.get_level_name(self.id, target_id)
    
    def learn_feedback(self, success: bool, topic: str = ""):
        """从反馈中学习"""
        if topic:
            if success:
                self.interest.update(topic, 0.1)
                self.consistency.record_success(topic)
            else:
                self.interest.update(topic, -0.05)
                self.consistency.record_failure(topic)
    
    def shutdown(self):
        """关闭魂核"""
        self._running = False
        self.memory.forget()
    
    def get_profile(self) -> Dict:
        """获取灵魂画像"""
        return {
            "id": self.id,
            "name": self.name,
            "traits": self.personality.traits,
            "emotions": self.emotion.emotions,
            "dominant_emotion": self.emotion.get_dominant(),
            "stats": self.stats,
            "memory_count": self.memory.stats["total"],
            "relationship_count": len(self.relationship.relationships.get(self.id, {})),
            "interests": self.interest.get_top_interests(5),
            "habits": self.habit.get_all_habits(),
            "curiosity_level": self.curiosity.curiosity_level,
            "intuition_accuracy": self.intuition.get_accuracy(),
            "dream_count": len(self.dream.dreams) if hasattr(self.dream, 'dreams') else 0,
            "coincidence_count": self.serendipity.get_coincidence_count(),
            "consistency_score": self.consistency.get_consistency_score(),
            "preferences": self.preference.preferences
        }
    
    def explain(self) -> str:
        """生成可读的灵魂画像"""
        lines = [f"=== {self.name}的灵魂 ===", ""]
        
        # 人格（前10个）
        lines.append("【人格】")
        traits_shown = sorted(self.personality.traits.items())[:10]
        for trait, value in traits_shown:
            bar = "█" * int(value * 10) + "░" * (10 - int(value * 10))
            lines.append(f"  {trait:12} {bar} {value:.2f}")
        
        # 情感
        lines.append("")
        lines.append("【情感】")
        dominant = self.emotion.get_dominant()
        for emotion, value in self.emotion.emotions.items():
            if value > 0.3:
                star = "★" if emotion == dominant else "·"
                lines.append(f"  {star} {emotion:12} {value:.2f}")
        
        # 兴趣
        interests = self.interest.get_top_interests(3)
        if interests:
            lines.append("")
            lines.append("【兴趣】")
            for i in interests:
                lines.append(f"  • {i['topic']}: {i['value']:.0%}")
        
        # 习惯
        habits = self.habit.get_all_habits()
        if habits:
            lines.append("")
            lines.append("【习惯】")
            for h in habits[:2]:
                lines.append(f"  • 每天{h['time']} {h['action']}（坚持{h['count']}天）")
        
        # 关系
        relations = self.relationship.get_network_summary(self.id, top_n=3)
        if relations:
            lines.append("")
            lines.append("【关系】")
            for r in relations:
                lines.append(f"  → {r['target']}: {r['level']} ({r['value']})")
        
        # 统计
        lines.append("")
        lines.append("【统计】")
        lines.append(f"  • 经历次数: {self.stats['experience_count']}")
        lines.append(f"  • 决策次数: {self.stats['decision_count']}")
        lines.append(f"  • 记忆数量: {self.memory.stats['total']}")
        lines.append(f"  • 好奇心: {self.curiosity.curiosity_level:.0%}")
        lines.append(f"  • 一致性: {self.consistency.get_consistency_score():.0%}")
        
        return "\n".join(lines)

3. corepersonality.py - 人格引擎
"""
人格引擎 - 魂核的基底
20个核心维度，从16PF/MMPI映射而来
使用心理学通用名称，安全可公开
"""

import time
import random
from typing import Dict, Optional


class PersonalityEngine:
    """
    人格引擎 - 管理角色的性格特质
    所有特质值范围 0-1
    """
    
    # 默认人格维度定义（心理学通用名称）
    DEFAULT_TRAITS = {
        # 情感类
        "warmth": 0.5,        # 温暖/亲和力
        "compassion": 0.5,    # 同情心
        "trust": 0.5,         # 信任他人
        "optimism": 0.5,      # 乐观
        
        # 行动类
        "energy": 0.5,        # 精力/活力
        "courage": 0.5,       # 勇气
        "patience": 0.5,      # 耐心
        "persistence": 0.5,   # 坚持/固执
        
        # 认知类
        "rationality": 0.5,   # 理性
        "curiosity": 0.5,     # 好奇心
        "creativity": 0.5,    # 创造力
        
        # 道德类
        "honesty": 0.5,       # 诚实
        "justice": 0.5,       # 正义感
        "loyalty": 0.5,       # 忠诚
        "selfishness": 0.5,   # 自私
        
        # 情绪倾向类
        "fear_tendency": 0.2,     # 恐惧倾向
        "anger_tendency": 0.2,    # 愤怒倾向
        "sadness_tendency": 0.2,  # 悲伤倾向
        "hope_tendency": 0.6,     # 希望倾向
        
        # 特殊
        "intuition": 0.5,      # 直觉力
        "resilience": 0.5,     # 韧性/恢复力
    }
    
    def __init__(self):
        self.traits = self.DEFAULT_TRAITS.copy()
        self.stability = 0.8          # 人格稳定性
        self.formed_at = time.time()
    
    def get_trait(self, name: str, default: float = 0.5) -> float:
        """获取特质值"""
        return self.traits.get(name, default)
    
    def set_trait(self, name: str, value: float):
        """设置特质值（0-1）"""
        if name in self.traits:
            self.traits[name] = max(0.0, min(1.0, value))
    
    def from_dict(self, data: Dict[str, float]):
        """从字典导入人格"""
        for key, value in data.items():
            if key in self.traits:
                self.traits[key] = max(0.0, min(1.0, value))
    
    def from_16pf(self, cattell_scores: Dict[str, float]):
        """
        从卡特尔16PF测试结果映射到人格维度
        16PF因子: A乐群性, B聪慧性, C稳定性, E恃强性,
                 F兴奋性, G有恒性, H敢为性, I敏感性,
                 L怀疑性, M幻想性, N世故性, O忧虑性,
                 Q1实验性, Q2独立性, Q3自律性, Q4紧张性
        """
        mappings = {
            "warmth": ["A", "H"],
            "compassion": ["I", "A"],
            "trust": ["A", "L"],
            "optimism": ["C", "O"],
            "energy": ["C", "Q4"],
            "courage": ["H", "I"],
            "patience": ["Q3", "Q4"],
            "persistence": ["G", "Q3"],
            "rationality": ["B", "M"],
            "curiosity": ["Q1", "M"],
            "honesty": ["G", "N"],
            "justice": ["G", "Q3"],
            "loyalty": ["A", "Q2"],
            "selfishness": ["E", "Q2"],
            "fear_tendency": ["O", "Q4"],
            "anger_tendency": ["E", "Q4"],
            "sadness_tendency": ["I", "O"],
        }
        
        reverse_factors = {"L", "O", "Q4", "I", "M", "N", "Q2"}
        
        for trait, factors in mappings.items():
            values = []
            for f in factors:
                if f in cattell_scores:
                    score = cattell_scores[f]
                    if f in reverse_factors:
                        score = 10 - score
                    values.append(score / 10)
            if values:
                self.traits[trait] = sum(values) / len(values)
        
        self._normalize()
    
    def _normalize(self):
        """确保所有值在 0-1 之间"""
        for k in self.traits:
            self.traits[k] = max(0.0, min(1.0, self.traits[k]))
    
    def describe(self) -> str:
        """生成人格描述文本"""
        traits = self.traits
        if traits["warmth"] > 0.7:
            return "温暖亲切的人"
        elif traits["courage"] > 0.7:
            return "勇敢果断的人"
        elif traits["compassion"] > 0.7:
            return "富有同情心的人"
        elif traits["optimism"] > 0.7:
            return "乐观开朗的人"
        else:
            return "性格温和的人"
    
    @staticmethod
    def create_preset(preset_name: str) -> "PersonalityEngine":
        """创建预设人格"""
        engine = PersonalityEngine()
        
        presets = {
            "kind": {"warmth": 0.8, "compassion": 0.8, "selfishness": 0.2},

class EmotionEngine:

        """根据事件更新情感"""
        changes = {}
        
        # 事件影响
        if event_type == "disaster":
            changes["fear"] = intensity * 0.3
            changes["anxiety"] = 0.1
            changes["hope"] = -0.1
        
        elif event_type == "help_others":
            changes["joy"] = 0.2
            changes["hope"] = 0.1
            changes["gratitude"] = 0.1
        
        elif event_type == "insult":
            changes["anger"] = 0.2
            changes["sadness"] = 0.1
        
        elif event_type == "gift":
            changes["joy"] = 0.2
            changes["gratitude"] = 0.2
        
        elif event_type == "betray":
            changes["sadness"] = 0.3
            changes["anger"] = 0.2
            changes["despair"] = 0.1

            changes["joy"] = 0.4
            changes["hope"] = 0.2

            changes["curiosity_emotion"] = 0.2
            changes["joy"] = 0.1
        
        elif event_type == "success":
            changes["pride"] = 0.3
            changes["joy"] = 0.2
        
        elif event_type == "failure":
            changes["shame"] = 0.2
            changes["sadness"] = 0.1
        
        elif event_type == "conversation":
            if content:
                if any(k in content for k in ["谢谢", "感谢"]):
                    changes["gratitude"] = 0.2
                    changes["joy"] = 0.1
                elif any(k in content for k in ["对不起", "抱歉"]):
                    changes["shame"] = 0.1
                elif any(k in content for k in ["想", "念"]):
                    changes["missing"] = 0.1
                elif any(k in content for k in ["厉害", "佩服"]):
                    changes["pride"] = 0.1
                    changes["joy"] = 0.1
        
        # 强度缩放
        for key in changes:
            changes[key] *= intensity
        
        # 人格调节（简单版）
        if personality:
            if personality.get("optimism", 0.5) > 0.7:
                changes["hope"] = changes.get("hope", 0) + 0.1
            if personality.get("fear_tendency", 0.2) > 0.6:
                changes["fear"] = changes.get("fear", 0) * 1.3
            if personality.get("anger_tendency", 0.2) > 0.6:

            if personality.get("hope_tendency", 0.6) > 0.7:
                changes["hope"] = changes.get("hope", 0) + 0.1

        for key, delta in changes.items():
            if key in self.emotions:
                self.emotions[key] = max(0.0, min(1.0, self.emotions[key] + delta))

        """情感自然衰减"""
        for emotion, rate in self.decay_rates.items():
            decay_amount = rate * (hours_passed / 24)
            self.emotions[emotion] = max(0.0, self.emotions[emotion] - decay_amount)
        
        # 复杂情感衰减更慢
        complex_emotions = ["missing", "loneliness", "gratitude", "hope", "pride", "shame"]
        for emotion in complex_emotions:
            self.emotions[emotion] = max(0.0, self.emotions[emotion] - 0.02 * (hours_passed / 24))
        
        self._normalize()

        return self.emotions.get(emotion, 0.0)
    
    def get_dominant(self) -> str:
        """获取主导情感"""
        return max(self.emotions.items(), key=lambda x: x[1])[0]
    
    def get_intensity(self) -> float:
        """获取整体情感强度"""
        return sum(self.emotions.values()) / len(self.emotions)
    
    def check_thresholds(self) -> List[str]:
        """检查是否触发情感阈值"""
        triggered = []
        for emotion, value in self.emotions.items():
            if emotion in self.thresholds and value > self.thresholds[emotion]:
                triggered.append(emotion)
        return triggered
    
    def _normalize(self):
        for k in self.emotions:
            self.emotions[k] = max(0.0, min(1.0, self.emotions[k]))

5. corememory.py - 记忆引擎
"""
记忆引擎 - 魂核的过去
固定衰减，不与人格挂钩
"""

import time
import uuid
from typing import Dict, List, Optional, Any
from dataclasses import dataclass, field


@dataclass
class Memory:
    """一段记忆"""
    id: str
    content: str
    type: str                    # event, person, place, skill, forced
    importance: int              # 1-10
    created_at: float
    last_accessed: Optional[float] = None
    access_count: int = 0
    strength: float = 1.0
    emotion: Optional[str] = None
    associations: List[str] = field(default_factory=list)
    
    def strengthen(self):
        """被回忆时强化"""
        self.access_count += 1
        self.last_accessed = time.time()
        self.strength = min(1.0, self.strength + 0.1)
    
    def decay(self, days_passed: float):
        """随时间衰减（固定速率，不与人格挂钩）"""
        if self.importance >= 9:
            return
        
        # 固定衰减率
        decay_rate = 0.05 * (10 - self.importance) / 9
        self.strength = max(0.0, self.strength * (1 - decay_rate * days_passed))
    
    def is_forgotten(self) -> bool:
        return self.strength < 0.1
    
    def to_dict(self) -> Dict:
        return {
            "id": self.id,
            "content": self.content,
            "type": self.type,
            "importance": self.importance,
            "strength": self.strength,
            "emotion": self.emotion,
        }


class MemoryEngine:
    """
    记忆引擎 - 存储、回忆、遗忘
    固定衰减，安全可公开
    """
    
    def __init__(self):
        self.memories = {
            "short_term": [],    # 重要性 1-4
            "long_term": [],     # 重要性 5-8
            "permanent": []      # 重要性 9-10
        }
        
        self._indices = {
            "by_type": {},
            "by_emotion": {},
        }
        
        self.stats = {
            "total": 0,
            "forgotten": 0,
        }
    
    def create_memory(self, content: str, type: str = "event",
                      importance: int = 5, emotion: str = None) -> Memory:
        """创建一段记忆"""
        return Memory(
            id=str(uuid.uuid4())[:8],
            content=content,
            type=type,
            importance=min(10, max(1, importance)),
            created_at=time.time(),
            emotion=emotion
        )
    
    def store(self, memory: Memory):
        """存储记忆"""
        if memory.importance >= 9:
            self.memories["permanent"].append(memory)
        elif memory.importance >= 5:
            self.memories["long_term"].append(memory)
        else:
            self.memories["short_term"].append(memory)
        
        # 更新索引
        if memory.type not in self._indices["by_type"]:
            self._indices["by_type"][memory.type] = []
        self._indices["by_type"][memory.type].append(memory.id)
        
        if memory.emotion:
            if memory.emotion not in self._indices["by_emotion"]:
                self._indices["by_emotion"][memory.emotion] = []
            self._indices["by_emotion"][memory.emotion].append(memory.id)
        
        self.stats["total"] += 1
    
    def store_forced(self, content: str, importance: int = 9, emotion: str = None) -> Memory:
        """强制存储为永久记忆"""
        memory = self.create_memory(content, "forced", importance, emotion)
        self.store(memory)
        return memory
    
    def recall(self, cue: str, context: Dict = None, limit: int = 5) -> List[Memory]:
        """根据线索回忆记忆"""
        results = []
        
        for category in ["permanent", "long_term", "short_term"]:
            for memory in self.memories[category]:
                relevance = self._calc_relevance(memory, cue, context)
                if relevance > 0.3:
                    results.append((relevance, memory))
                    memory.strengthen()
        
        results.sort(key=lambda x: -x[0])
        return [mem for _, mem in results[:limit]]
    
    def _calc_relevance(self, memory: Memory, cue: str, context: Dict = None) -> float:
        """计算记忆与线索的相关性"""
        score = 0.0
        
        if cue and cue.lower() in memory.content.lower():
            score += 0.4
        
        if context and context.get("type") == memory.type:
            score += 0.3
        
        if context and context.get("emotion") == memory.emotion:
            score += 0.2
        
        score += memory.strength * 0.2
        score += memory.importance / 50
        
        return min(1.0, score)
    
    def get_strongest(self, limit: int = 5) -> List[Memory]:
        """获取最强的记忆"""
        all_memories = (self.memories["permanent"] + 
                        self.memories["long_term"] + 
                        self.memories["short_term"])
        scored = [(mem.importance / 10 + mem.strength, mem) for mem in all_memories]
        scored.sort(key=lambda x: -x[0])
        return [mem for _, mem in scored[:limit]]
    
    def get_oldest(self, limit: int = 5) -> List[Memory]:
        """获取最老的记忆"""
        all_memories = (self.memories["permanent"] + 
                        self.memories["long_term"] + 
                        self.memories["short_term"])
        all_memories.sort(key=lambda x: x.created_at)
        return all_memories[:limit]
    
    def get_recent(self, limit: int = 10) -> List[Memory]:
        """获取最近的记忆"""
        all_memories = (self.memories["permanent"] + 
                        self.memories["long_term"] + 
                        self.memories["short_term"])
        all_memories.sort(key=lambda x: x.created_at, reverse=True)
        return all_memories[:limit]
    
    def forget(self):
        """执行遗忘（每日调用）"""
        today = time.time()
        
        for category in ["short_term", "long_term"]:
            to_remove = []
            for memory in self.memories[category]:
                last_access = memory.last_accessed or memory.created_at
                days_passed = (today - last_access) / 86400
                memory.decay(days_passed)
                if memory.is_forgotten():
                    to_remove.append(memory)
            
            for memory in to_remove:
                self.memories[category].remove(memory)
                self.stats["forgotten"] += 1
                self.stats["total"] -= 1
    
    def strengthen_memory(self, memory_id: str):
        """强化指定记忆"""
        for category in self.memories.values():
            for memory in category:
                if memory.id == memory_id:
                    memory.strengthen()
                    return True
        return False

"""

import random
import time
from typing import Dict, List, Any, Optional, Tuple


class DecisionEngine:
    """
    游戏版决策引擎
    
    特点：
    - 加入资源因子（与论文差异化）
    - 权重分配不同
    - 无需求层次
    """
    
    def __init__(self, soul):
        self.soul = soul
        
        # 决策权重
        self.weights = {
            "personality": 0.25,
            "emotion": 0.25,
            "memory": 0.20,
            "resource": 0.30,
        }

        self.action_mappings = {
            "help": {
                "thresholds": [0.6],
                "actions": ["help_others", "self_first"]
            },
            "conversation": {
                "thresholds": [0.7, 0.4],
                "actions": ["open_up", "listen", "deflect", "silence"]
            },
            "evacuation": {
                "thresholds": [0.7, 0.5, 0.3],
                "actions": ["leave_now", "prepare", "wait", "stay"]
            },
            "rescue": {
                "thresholds": [0.7, 0.5],
                "actions": ["go_rescue", "call_help", "stay_safe"]
            },
            "learn": {
                "thresholds": [0.5],
                "actions": ["learn", "ignore"]
            },
            "default": {
                "thresholds": [0.6],
                "actions": ["proceed", "wait", "avoid"]
            }
        }
    
    def decide(self, situation: Dict[str, Any], memories: List = None) -> Dict[str, Any]:
        """做出决策"""
        sit_type = situation.get("type", "default")
        
        personality_score = self._score_personality(sit_type, situation)
        emotion_score = self._score_emotion(sit_type, situation)
        memory_score = self._score_memory(sit_type, situation, memories)
        resource_score = self._score_resource(sit_type, situation)
        
        final_score = (
            personality_score * self.weights["personality"] +
            emotion_score * self.weights["emotion"] +
            memory_score * self.weights["memory"] +
            resource_score * self.weights["resource"]
        )
        
        final_score += random.uniform(-0.1, 0.1)
        final_score = max(0.0, min(1.0, final_score))
        
        action, confidence = self._map_to_action(final_score, sit_type)
        explanation = self._generate_explanation(action, final_score, sit_type)
        
        self.history.append({
            "time": time.time(),
            "situation": sit_type,
            "action": action,
            "score": final_score,
            "personality": personality_score,
            "emotion": emotion_score,
            "memory": memory_score,
            "resource": resource_score
        })
        
        return {
            "action": action,
            "explanation": explanation,
            "confidence": confidence,
            "score": final_score
        }
    
    def _score_personality(self, sit_type: str, situation: Dict) -> float:
        """人格评分"""
        traits = self.soul.personality.traits
        score = 0.5
        
        if sit_type == "help":
            if traits.get("compassion", 0.5) > 0.6:
                score += 0.2
            if traits.get("selfishness", 0.5) > 0.6:
                score -= 0.2
        
        elif sit_type == "conversation":
            if traits.get("warmth", 0.5) > 0.6:
                score += 0.2
            if traits.get("trust", 0.5) > 0.6:
                score += 0.1
        
        elif sit_type == "evacuation":
            if traits.get("courage", 0.5) > 0.6:
                score += 0.1
            if traits.get("fear_tendency", 0.2) > 0.6:
                score -= 0.1
            if traits.get("persistence", 0.5) > 0.7:
                score += 0.1
        
        elif sit_type == "rescue":
            if traits.get("courage", 0.5) > 0.7:
                score += 0.2
            if traits.get("compassion", 0.5) > 0.6:

    def _score_emotion(self, sit_type: str, situation: Dict) -> float:

            "learn": "学一下", "ignore": "先不管",
            "proceed": "继续", "avoid": "避开"

        return f"{name}决定{action_desc}，因为{'、'.join(reasons[:2])}"

class RelationshipEngine:

    - 人格影响：使用通用维度（compassion, selfishness）
    - 调节逻辑：简单加减法（if > 0.7: delta += 1）
    """
    
    # 事件类型对应的基础变化
    DEFAULT_EVENTS = {
        "help": 1, "rescue": 2, "gift": 1, "praise": 1, "keep_promise": 1,
        "betray": -2, "insult": -1, "lie": -1, "break_promise": -2, "ignore_help": -1,
        "rob": -2, "fight": -2, "share": 1, "protect": 2,
        "teach": 1, "learn_from": 1, "disagree": -1, "agree": 1,
    }
    
    # 阈值触发事件
    THRESHOLDS = {
        -8: "仇恨觉醒", -5: "敌意显露", -3: "反感",
        3: "友善", 5: "信赖", 8: "挚友", 10: "灵魂羁绊"
    }
    
    def __init__(self, enable_personality_effect: bool = True):
        self.relationships: Dict[str, Dict[str, Relationship]] = {}
        self.events = self.DEFAULT_EVENTS.copy()
        self.enable_personality_effect = enable_personality_effect
        self.stats = {"total_changes": 0, "threshold_triggers": 0}
    
    def get(self, agent_id: str, target_id: str) -> Relationship:
        """获取或创建关系"""
        if agent_id not in self.relationships:
            self.relationships[agent_id] = {}
        if target_id not in self.relationships[agent_id]:
            self.relationships[agent_id][target_id] = Relationship(target_id=target_id)
        return self.relationships[agent_id][target_id]
    
    def get_value(self, agent_id: str, target_id: str) -> int:
        return self.get(agent_id, target_id).value
    
    def get_level_name(self, agent_id: str, target_id: str) -> str:
        return self.get(agent_id, target_id).get_level_name()
    
    def change(self, agent_id: str, target_id: str, delta: int, 
               reason: str = "", personality: Optional[Dict] = None) -> Dict:
        """
        改变关系值
        
        L1安全级人格调节：
        - 使用通用维度 compassion, selfishness
        - 简单加减法（if > 0.7: delta += 1）
        """
        rel = self.get(agent_id, target_id)
        old_value = rel.value
        
        adjusted_delta = delta
        
        if self.enable_personality_effect and personality:
            if delta > 0:
                compassion = personality.get("compassion", 0.5)
                if compassion > 0.7:
                    adjusted_delta += 1
                elif compassion < 0.3:
                    adjusted_delta -= 1
            
            if delta < 0:
                selfishness = personality.get("selfishness", 0.5)
                if selfishness > 0.7:
                    adjusted_delta -= 1
                elif selfishness < 0.3:
                    adjusted_delta += 1
        
        new_value = max(-10, min(10, old_value + adjusted_delta))
        rel.value = new_value
        
        change_record = {
            "delta": delta, "adjusted": adjusted_delta,
            "old": old_value, "new": new_value, "reason": reason
        }
        rel.history.append(change_record)
        if len(rel.history) > 20:
            rel.history.pop(0)
        rel.last_change = change_record
        
        self.stats["total_changes"] += 1
        
        triggered = None
        for threshold, event_name in sorted(self.THRESHOLDS.items()):
            if old_value < threshold <= new_value:
                triggered = event_name
                self.stats["threshold_triggers"] += 1
                break
        
        return {"old": old_value, "new": new_value, "delta": adjusted_delta, "trigger": triggered}
    
    def apply_event(self, agent_id: str, target_id: str, event_type: str,
                    personality: Optional[Dict] = None) -> Dict:
        """根据事件类型改变关系"""
        delta = self.events.get(event_type, 0)
        if delta == 0:
            return {"old": self.get_value(agent_id, target_id), 
                    "new": self.get_value(agent_id, target_id), "delta": 0}
        return self.change(agent_id, target_id, delta, f"事件:{event_type}", personality)
    
    def get_network_summary(self, agent_id: str, top_n: int = 5) -> List[Dict]:
        """获取关系网络摘要"""
        if agent_id not in self.relationships:
            return []
        
        relations = [
            {"target": tid, "value": rel.value, "level": rel.get_level_name()}
            for tid, rel in self.relationships[agent_id].items()
            if rel.value != 0
        ]
        relations.sort(key=lambda x: -abs(x["value"]))
        return relations[:top_n]
    
    def set_initial(self, agent_id: str, target_id: str, value: int):
        """设置初始关系"""
        rel = self.get(agent_id, target_id)
        rel.value = max(-10, min(10, value))
    
    def daily_update(self):
        """每日更新（关系自然衰减）"""
        for agent_id, relations in self.relationships.items():
            for target_id, rel in list(relations.items()):
                if rel.value > 0 and random.random() < 0.1:
                    rel.value = max(0, rel.value - 1)
                elif rel.value < 0 and random.random() < 0.05:
                    rel.value = min(0, rel.value + 1)

@dataclass
class ResourceProfile:

    gold: float = 50.0
    wood: float = 20.0
    stone: float = 30.0
    energy: float = 80.0


class ResourceSystem:
    """
    资源管理系统
    """
    
    def __init__(self):
        self.resources: Dict[str, ResourceProfile] = {}
        self.default_resources = ResourceProfile()
    
    def register(self, agent_id: str, initial: Optional[ResourceProfile] = None):
        """注册智能体资源"""
        self.resources[agent_id] = initial if initial else ResourceProfile()
    
    def get(self, agent_id: str, resource_type: str = "food") -> float:
        profile = self.resources.get(agent_id)
        if not profile:
            return 0
        return getattr(profile, resource_type, 0)
    
    def set(self, agent_id: str, resource_type: str, value: float):
        profile = self.resources.get(agent_id)
        if profile:
            setattr(profile, resource_type, max(0, value))
    
    def add(self, agent_id: str, resource_type: str, amount: float) -> float:
        current = self.get(agent_id, resource_type)
        new_value = current + amount
        self.set(agent_id, resource_type, new_value)
        return new_value
    
    def remove(self, agent_id: str, resource_type: str, amount: float) -> bool:
        current = self.get(agent_id, resource_type)
        if current >= amount:
            self.set(agent_id, resource_type, current - amount)
            return True
        return False
    
    def transfer(self, from_id: str, to_id: str, resource_type: str, amount: float) -> bool:
        if self.remove(from_id, resource_type, amount):
            self.add(to_id, resource_type, amount)
            return True
        return False
    
    def is_scarce(self, agent_id: str, resource_type: str = "food", threshold: float = 30) -> bool:
        return self.get(agent_id, resource_type) < threshold
    
    def get_scarcity_level(self, agent_id: str, resource_type: str = "food") -> float:
        current = self.get(agent_id, resource_type)
        if current >= 100:
            return 0
        if current <= 0:
            return 1
        return max(0, min(1, (100 - current) / 100))
    
    def get_resource_factor(self, agent_id: str) -> float:
        scarcity = self.get_scarcity_level(agent_id, "food")
        return 1.0 - scarcity * 0.5
    
    def daily_update(self):
        """每日资源消耗"""
        for agent_id, profile in self.resources.items():
            profile.food = max(0, profile.food - 2)
            profile.energy = max(0, profile.energy - 1)

"""
空间系统 - 游戏版
"""

import random
import math
from typing import Dict, List, Optional, Tuple
from dataclasses import dataclass


@dataclass
class Position:
    """位置坐标"""
    x: float
    y: float
    
    def distance_to(self, other: 'Position') -> float:
        return math.sqrt((self.x - other.x)**2 + (self.y - other.y)**2)


class SpatialSystem:
    """
    空间管理系统
    """
    
    def __init__(self, width: float = 100, height: float = 100):
        self.width = width
        self.height = height
        self.positions: Dict[str, Position] = {}
        self.interaction_range: float = 15.0
    
    def register(self, agent_id: str, x: Optional[float] = None, y: Optional[float] = None):
        if x is None:
            x = random.uniform(0, self.width)
        if y is None:
            y = random.uniform(0, self.height)
        self.positions[agent_id] = Position(x, y)
    
    def get_position(self, agent_id: str) -> Optional[Position]:
        return self.positions.get(agent_id)
    
    def move_random(self, agent_id: str, step: float = 5.0):
        pos = self.positions.get(agent_id)
        if pos:
            new_x = pos.x + random.uniform(-step, step)
            new_y = pos.y + random.uniform(-step, step)
            pos.x = max(0, min(self.width, new_x))
            pos.y = max(0, min(self.height, new_y))
    
    def move_towards(self, agent_id: str, target_id: str, step: float = 5.0):
        """向目标移动"""
        pos = self.positions.get(agent_id)
        target_pos = self.positions.get(target_id)
        if pos and target_pos:
            dx = target_pos.x - pos.x
            dy = target_pos.y - pos.y
            dist = math.sqrt(dx*dx + dy*dy)
            if dist > 0:
                move = min(step, dist)
                pos.x += (dx / dist) * move
                pos.y += (dy / dist) * move
    
    def distance(self, a_id: str, b_id: str) -> float:
        a_pos = self.positions.get(a_id)
        b_pos = self.positions.get(b_id)
        if not a_pos or not b_pos:
            return float('inf')
        return a_pos.distance_to(b_pos)
    
    def can_interact(self, a_id: str, b_id: str, range_multiplier: float = 1.0) -> bool:
        return self.distance(a_id, b_id) <= self.interaction_range * range_multiplier
    
    def get_nearby(self, agent_id: str, radius: Optional[float] = None) -> List[str]:
        if radius is None:
            radius = self.interaction_range
        
        nearby = []
        pos = self.positions.get(agent_id)
        if not pos:
            return nearby
        
        for other_id, other_pos in self.positions.items():
            if other_id != agent_id and pos.distance_to(other_pos) <= radius:
                nearby.append(other_id)
        return nearby

import random
from typing import Dict, List, Optional, Callable, Any
from dataclasses import dataclass, field


@dataclass
class GameEvent:
    """游戏事件"""
    name: str
    description: str
    probability: float
    duration: int = 1
    effects: Dict[str, Any] = field(default_factory=dict)
    
    def __post_init__(self):
        self.remaining_days = 0
        self.active = False
    
    def trigger(self):
        self.active = True
        self.remaining_days = self.duration
    
    def update(self):
        if self.active:
            self.remaining_days -= 1
            if self.remaining_days <= 0:
                self.active = False


class EventSystem:
    """
    事件管理系统
    """
    
    def __init__(self):
        self.events: List[GameEvent] = []
        self.active_events: List[GameEvent] = []
        self.event_callbacks: Dict[str, List[Callable]] = {}
        
        self._register_default_events()
    
    def _register_default_events(self):
        """注册默认事件"""
        self.events = [
            GameEvent("famine", "饥荒来袭，食物变得稀缺", 0.05, 5,
                      {"food_multiplier": 0.5, "food_consumption": 2.0}),
            GameEvent("harvest", "丰收季节，食物充足", 0.10, 3,
                      {"food_multiplier": 1.5, "food_consumption": 0.5}),
            GameEvent("festival", "节日庆典，人们心情愉悦", 0.08, 1,
                      {"relation_bonus": 0.5, "happiness_bonus": 0.3}),
            GameEvent("raiders", "掠夺者袭击，资源受损", 0.03, 2,
                      {"food_loss": 0.2, "gold_loss": 0.3, "danger_level": 0.7}),
            GameEvent("storm", "暴风雨来袭，出行不便", 0.06, 2,
                      {"movement_penalty": 0.5, "damage_risk": 0.2}),
            GameEvent("discovery", "发现新资源", 0.04, 1,
                      {"food_gain": 30, "gold_gain": 20}),
            GameEvent("celebration", "庆祝活动，关系改善", 0.07, 1,
                      {"relation_bonus": 0.3, "happiness_bonus": 0.2}),
        ]
    
    def add_event(self, event: GameEvent):
        """添加自定义事件"""
        self.events.append(event)
    
    def on_event(self, event_name: str, callback: Callable):
        """注册事件回调"""
        if event_name not in self.event_callbacks:
            self.event_callbacks[event_name] = []
        self.event_callbacks[event_name].append(callback)
    
    def update(self, world) -> Optional[GameEvent]:
        """每日更新，触发事件"""
        for event in self.active_events[:]:
            event.update()
            if not event.active:
                self.active_events.remove(event)
        
        for event in self.events:
            if not event.active and random.random() < event.probability:
                event.trigger()
                self.active_events.append(event)
                
                if event.name in self.event_callbacks:
                    for callback in self.event_callbacks[event.name]:
                        callback(world, event)
                
                return event
        
        return None
    
    def get_active_events(self) -> List[GameEvent]:
        return [e for e in self.active_events if e.active]
    
    def get_interaction_modifier(self) -> float:
        modifier = 1.0
        for event in self.active_events:
            if "interaction_penalty" in event.effects:
                modifier += event.effects["interaction_penalty"]
        return max(0.1, modifier)

        })

        if len(self.interest_history) > 1000:

        )

            strength=0.1

            "chosen": chosen_action,
            "expected": expected,
            "consistent": is_consistent,
            "timestamp": time.time()
        })
        
        if not is_consistent:
            self.inconsistencies.append({
                "situation": situation.get("type", "unknown"),
                "chosen": chosen_action,
                "expected": expected,
                "timestamp": time.time()
            })
        
        # 限制历史长度
        if len(self.behavior_history) > 1000:
            self.behavior_history = self.behavior_history[-500:]
        if len(self.inconsistencies) > 200:
            self.inconsistencies = self.inconsistencies[-100:]
    
    def _predict_expected_behavior(self, situation: Dict) -> str:
        """基于人格预测应该的行为"""
        traits = self.soul.personality.traits
        sit_type = situation.get("type", "default")
        
        # 简单的预测逻辑
        if sit_type == "help":
            if traits.get("compassion", 0.5) > 0.5:
                return "help_others"
            else:
                return "self_first"
        
        elif sit_type == "conversation":
            if traits.get("warmth", 0.5) > 0.5:
                return "open_up"
            else:
                return "listen"
        
        elif sit_type == "evacuation":
            if traits.get("courage", 0.5) > 0.5:
                return "leave_now"
            else:
                return "wait"
        
        elif sit_type == "rescue":
            if traits.get("courage", 0.5) > 0.5 and traits.get("compassion", 0.5) > 0.5:
                return "go_rescue"
            else:
                return "call_help"
        
        return "proceed"
    
    def get_consistency_score(self) -> float:
        """获取一致性评分"""
        if not self.behavior_history:
            return 0.8  # 默认高分
        
        recent = self.behavior_history[-50:]
        consistent = sum(1 for b in recent if b["consistent"])
        return consistent / len(recent)
    
    def record_success(self, topic: str):
        """记录成功"""
        self.success_count[topic] += 1
    
    def record_failure(self, topic: str):
        """记录失败"""
        self.failure_count[topic] += 1
    
    def get_success_rate(self, topic: str = None) -> float:
        """获取成功率"""
        if topic:
            total = self.success_count[topic] + self.failure_count[topic]
            if total == 0:
                return 0.5
            return self.success_count[topic] / total
        
        total_success = sum(self.success_count.values())
        total_failure = sum(self.failure_count.values())
        total = total_success + total_failure
        if total == 0:
            return 0.5
        return total_success / total
    
    def get_recent_inconsistencies(self, limit: int = 5) -> List[Dict]:
        """获取最近的不一致行为"""
        return self.inconsistencies[-limit:]

"""
世界管理器 - 整合所有系统
"""

import random
from typing import Dict, List, Optional

from .coresoul import Soul, PerceptionContext
from .coreresource import ResourceSystem
from .corespatial import SpatialSystem
from .coreevent import EventSystem


class World:
    """
    游戏世界管理器
    """
    
    def __init__(self, width: float = 100, height: float = 100):
        self.agents: Dict[str, Soul] = {}
        self.resource_system = ResourceSystem()
        self.spatial_system = SpatialSystem(width, height)
        self.event_system = EventSystem()
        
        self.day = 0
        self.history: List[Dict] = []
        
        self._register_event_callbacks()
    
    def _register_event_callbacks(self):
        """注册事件回调"""
        def on_famine(world, event):
            for agent in world.agents.values():
                world.resource_system.add(agent.id, "food", -30)
                # 情感影响
                agent.emotion.add_emotion("fear", 0.2)
                agent.emotion.add_emotion("anxiety", 0.1)
        
        def on_festival(world, event):
            for _ in range(10):
                if len(world.agents) >= 2:
                    a_id, b_id = random.sample(list(world.agents.keys()), 2)
                    world.agents[a_id].update_relationship(b_id, "praise")
                    # 情感影响
                    world.agents[a_id].emotion.add_emotion("joy", 0.1)
        
        def on_discovery(world, event):
            for agent in world.agents.values():
                world.resource_system.add(agent.id, "food", 15)
                agent.emotion.add_emotion("joy", 0.15)
                agent.emotion.add_emotion("curiosity_emotion", 0.2)
        
        self.event_system.on_event("famine", on_famine)
        self.event_system.on_event("festival", on_festival)
        self.event_system.on_event("discovery", on_discovery)
    
    def add_agent(self, agent: Soul, x: Optional[float] = None, y: Optional[float] = None):
        """添加智能体"""
        self.agents[agent.id] = agent
        
        agent.set_resource_system(self.resource_system)
        agent.set_spatial_system(self.spatial_system)
        agent.set_event_system(self.event_system)
        
        self.resource_system.register(agent.id)
        self.spatial_system.register(agent.id, x, y)
    
    def update(self):
        """更新世界（一天）"""
        self.day += 1
        
        triggered_event = self.event_system.update(self)
        
        for agent in self.agents.values():
            agent.daily_update()
        
        self._daily_consumption()
        self._random_interactions(10)
        self._record_history(triggered_event)
    
    def _daily_consumption(self):
        """每日资源消耗"""
        for agent_id in self.agents:
            self.resource_system.add(agent_id, "food", -2)
            self.resource_system.add(agent_id, "energy", -1)
    
    def _random_interactions(self, count: int):
        """随机配对交互"""
        agent_ids = list(self.agents.keys())
        if len(agent_ids) < 2:
            return
        
        for _ in range(count):
            a_id, b_id = random.sample(agent_ids, 2)
            a = self.agents[a_id]
            b = self.agents[b_id]
            
            if not self.spatial_system.can_interact(a_id, b_id):
                continue
            
            event = self._determine_interaction(a_id, b_id)
            
            ctx = PerceptionContext(
                event_type=event,
                content=f"{a.name}对{b.name}执行了{event}",
                intensity=0.5,
                target_id=b_id
            )
            
            decision = a.perceive(ctx)
            
            # 更新关系
            if event in ["help", "gift", "praise", "teach"]:
                a.update_relationship(b_id, event)
            elif event in ["insult", "betray", "ignore_help", "disagree"]:
                a.update_relationship(b_id, event)
            
            # 资源转移
            if event == "gift":
                self.resource_system.transfer(a_id, b_id, "food", 10)
            elif event == "rob":
                self.resource_system.transfer(b_id, a_id, "food", 15)
            elif event == "share":
                self.resource_system.transfer(a_id, b_id, "food", 5)
    
    def _determine_interaction(self, a_id: str, b_id: str) -> str:
        """确定交互类型"""
        # 资源稀缺影响
        if self.resource_system.is_scarce(a_id, "food", 30):
            return random.choices(
                ["rob", "help", "gift", "insult", "praise", "share"],
                weights=[0.4, 0.1, 0.05, 0.2, 0.1, 0.15]
            )[0]
        
        # 关系影响
        a = self.agents[a_id]
        relation = a.get_relation(b_id)
        
        if relation >= 5:
            return random.choices(
                ["help", "gift", "praise", "teach", "share"],
                weights=[0.3, 0.25, 0.2, 0.15, 0.1]
            )[0]
        elif relation <= -3:
            return random.choices(
                ["insult", "betray", "ignore_help", "disagree"],
                weights=[0.4, 0.3, 0.2, 0.1]
            )[0]
        
        return random.choice(["help", "gift", "insult", "praise", "conversation", "share"])
    
    def _record_history(self, event):
        """记录历史"""
        self.history.append({
            "day": self.day,
            "event": event.name if event else None,
            "agent_count": len(self.agents)
        })
    
    def get_statistics(self) -> Dict:
        """获取世界统计"""
        total_relations = 0
        positive_relations = 0
        for agent in self.agents.values():
            relations = agent.relationship.get_network_summary(agent.id, top_n=100)
            total_relations += len(relations)
            positive_relations += sum(1 for r in relations if r["value"] > 0)
        
        # 计算平均好奇心
        avg_curiosity = sum(a.curiosity.curiosity_level for a in self.agents.values()) / max(len(self.agents), 1)
        
        return {
            "day": self.day,
            "agent_count": len(self.agents),
            "total_relations": total_relations,
            "positive_rate": positive_relations / total_relations if total_relations > 0 else 0,
            "active_events": [e.name for e in self.event_system.get_active_events()],
            "avg_curiosity": avg_curiosity,
            "total_interactions": sum(a.stats["decision_count"] for a in self.agents.values())
        }
